63  size_t end_iteration(const VecRef<R>& parameters, const VecRef<R>& action) override {
64  auto prof = this->profiler();
65  auto m_prof = prof->push("itsolv::end_iteration");
66  if (m_dspace_resetter.do_reset(this->m_stats->iterations, this->m_xspace->dimensions())) {
67  m_resetting_in_progress = true;
68  this->m_working_set = m_dspace_resetter.run(parameters, *this->m_xspace, this->m_subspace_solver->solutions(),
69  propose_rspace_norm_thresh, propose_rspace_svd_thresh,
70  *this->m_handlers, *this->m_logger);
71  } else {
72  m_resetting_in_progress = false;
73  prof->start("end_iteration (propose_rspace)");
74  this->m_working_set = detail::propose_rspace(
75  *this, parameters, action, *this->m_xspace, *this->m_subspace_solver, *this->m_handlers, *this->m_logger,
76  propose_rspace_svd_thresh, propose_rspace_norm_thresh, m_max_size_qspace, *this->profiler().get());
77  prof->stop();
78  }
79  this->m_stats->iterations++;
80  read_handler_counts(this->m_stats, this->m_handlers);

106  auto current_values = this->m_subspace_solver->eigenvalues();
107  this->m_value_errors.assign(current_values.size(), std::numeric_limits<scalar_type>::max());
108  for (size_t i = 0; i < std::min(m_last_values.size(), current_values.size()); i++)
