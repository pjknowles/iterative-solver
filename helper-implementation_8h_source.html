131  throw std::runtime_error("Size of eigenvectors/eigenvlaues do not match dimension!");

137  // static const char store_upper_triangle = 'U';

168  std::vector<double> eigvecs(dimension * dimension);
169  std::vector<double> eigvals(dimension);
170 
171  // call to lapack
172  int success = eigensolver_lapacke_dsyev(matrix, eigvecs, eigvals, dimension);
173  if (success < 0) {
174  throw std::invalid_argument("Invalid argument of eigensolver_lapacke_dsyev: ");
175  }
176  if (success > 0) {
177  throw std::runtime_error("Lapacke_dsyev (eigensolver) failed to converge. "
178  " elements of an intermediate tridiagonal form did not converge to zero.");
179  }

181  auto eigensystem = std::list<SVD<double>>{};
182 
183  // populate eigensystem
184  for (int i = dimension - 1; i >= 0;
185  i--) { // note: flipping this axis gives parity with results of eigen::jacobiSVD
186  auto temp_eigenproblem = SVD<double>{};
187  temp_eigenproblem.value = eigvals[i];
188  for (size_t j = 0; j < dimension; j++) {
189  temp_eigenproblem.v.emplace_back(eigvecs[j + (dimension * i)]);
190  }
191  eigensystem.emplace_back(temp_eigenproblem);
192  }
193 
194  return eigensystem;

222 size_t get_rank(std::vector<value_type> eigenvalues, value_type threshold) {
223  if (eigenvalues.size() == 0) {
224  return 0;
225  }

227  value_type threshold_scaled = threshold * max;

230  return count;

240 template <typename value_type>
241 size_t get_rank(std::list<SVD<value_type>> svd_system, value_type threshold) {
242  // compute max
243  value_type max_value = 0;
244  std::list<SVD<double>>::iterator it;
245  for (it = svd_system.begin(); it != svd_system.end(); it++) {
246  if (it->value > max_value) {
247  max_value = it->value;
248  }
249  }
250  // scale threshold
251  value_type threshold_scaled = threshold * max_value;
252 
253  size_t rank = 0;
254  // get rank
255  for (it = svd_system.begin(); it != svd_system.end(); it++) {
256  if (it->value > threshold_scaled) {
257  rank += 1;
258  }
259  }
260  return rank;
261 }

546 template <typename value_type, typename std::enable_if_t<!is_complex<value_type>{}, std::nullptr_t>>
547 void solve_LinearEquations(std::vector<value_type>& solution, std::vector<value_type>& eigenvalues,
548  const std::vector<value_type>& matrix, const std::vector<value_type>& metric,
549  const std::vector<value_type>& rhs, const size_t dimension, size_t nroot,
550  double augmented_hessian, double svdThreshold, int verbosity) {
551  const Eigen::Index nX = dimension;
552  solution.resize(nX * nroot);
553  // std::cout << "augmented_hessian "<<augmented_hessian<<std::endl;
554  if (augmented_hessian > 0) { // Augmented hessian
555  Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic> subspaceMatrix;
556  Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic> subspaceOverlap;
557  subspaceMatrix.conservativeResize(nX + 1, nX + 1);
558  subspaceOverlap.conservativeResize(nX + 1, nX + 1);
559  subspaceMatrix.block(0, 0, nX, nX) =
560  Eigen::Map<const Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic>>(matrix.data(), nX, nX);
561  subspaceOverlap.block(0, 0, nX, nX) =
562  Eigen::Map<const Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic>>(metric.data(), nX, nX);
563  eigenvalues.resize(nroot);
564  for (size_t root = 0; root < nroot; root++) {
565  for (Eigen::Index i = 0; i < nX; i++) {
566  subspaceMatrix(i, nX) = subspaceMatrix(nX, i) = -augmented_hessian * rhs[i + nX * root];
567  subspaceOverlap(i, nX) = subspaceOverlap(nX, i) = 0;
568  }
569  subspaceMatrix(nX, nX) = 0;
570  subspaceOverlap(nX, nX) = 1;
571  // std::cout << "subspace augmented hessian subspaceMatrix\n"<<subspaceMatrix<<std::endl;
572  // std::cout << "subspace augmented hessian subspaceOverlap\n"<<subspaceOverlap<<std::endl;
573 
574  Eigen::GeneralizedEigenSolver<Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic>> s(subspaceMatrix,
575  subspaceOverlap);
576  auto eval = s.eigenvalues();
577  auto evec = s.eigenvectors();
578  Eigen::Index imax = 0;
579  for (Eigen::Index i = 0; i < nX + 1; i++)
580  if (eval(i).real() < eval(imax).real())
581  imax = i;
582  eigenvalues[root] = eval(imax).real();
583  auto Solution = evec.col(imax).real().head(nX) / (augmented_hessian * evec.real()(nX, imax));
584  for (auto k = 0; k < nX; k++)
585  solution[k + nX * root] = Solution(k);
586  // std::cout << "subspace augmented hessian solution\n"<<Solution<<std::endl;
587  }
588  } else { // straight solution of linear equations
589  Eigen::Map<const Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> subspaceMatrixR(
590  matrix.data(), nX, nX);
591  Eigen::Map<const Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> RHS_R(rhs.data(), nX,
592  nroot);
593  Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic> subspaceMatrix = subspaceMatrixR;
594  Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic> RHS = RHS_R;
595  Eigen::Matrix<value_type, Eigen::Dynamic, Eigen::Dynamic> Solution;
596  Solution = subspaceMatrix.householderQr().solve(RHS);
597  // std::cout << "subspace linear equations solution\n"<<Solution<<std::endl;
598  for (size_t root = 0; root < nroot; root++)
599  for (auto k = 0; k < nX; k++)
600  solution[k + nX * root] = Solution(k, root);
601  }
602 }
