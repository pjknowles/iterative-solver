172  void construct_residual(const std::vector<int>& roots, const CVecRef<R>& params, const VecRef<R>& actions) override {
173  assert(params.size() >= roots.size());
174  const auto& norm = std::dynamic_pointer_cast<subspace::XSpace<R, Q, P>>(this->m_xspace)->rhs_norm();
175  for (size_t i = 0; i < roots.size(); ++i) {
176  const auto ii = roots[i];
177  this->m_handlers->rq().axpy(-1, rhs().at(ii), actions.at(i));
178  if (norm.at(ii) != 0) {
179  auto scal = 1 / norm[ii];
180  this->m_handlers->rr().scal(scal, actions.at(i));
181  }
182  }
183  }
