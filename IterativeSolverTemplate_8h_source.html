21 inline std::vector<std::pair<size_t, size_t>> parameter_batches(const size_t nsol, const size_t nparam) {
22  auto batches = std::vector<std::pair<size_t, size_t>>{};
23  if (nparam && nsol) {
24  auto n_batch = nsol / nparam + (nsol % nparam ? 1 : 0);
25  for (size_t ib = 0, start_sol = 0, end_sol = 0; ib < n_batch; ++ib, start_sol = end_sol) {
26  end_sol = std::min(start_sol + nparam, nsol);
27  batches.emplace_back(start_sol, end_sol);
28  }
29  }
30  return batches;
31 }

33 template <class R, class Q, class P>
34 void construct_solution(const VecRef<R>& params, const std::vector<int>& roots,
35  const subspace::Matrix<double>& solutions,
36  const std::vector<std::reference_wrapper<P>>& pparams,
37  const std::vector<std::reference_wrapper<Q>>& qparams,
38  const std::vector<std::reference_wrapper<Q>>& dparams, size_t oP, size_t oQ, size_t oD,
39  ArrayHandlers<R, Q, P>& handlers) {
40  auto prof = molpro::Profiler::single();
41  prof->start("get rd_mat");
42  assert(params.size() >= roots.size());
43  for (size_t i = 0; i < roots.size(); ++i) {
44  handlers.rr().fill(0, params.at(i));
45  }
46  subspace::Matrix<double> rp_mat(std::make_pair(pparams.size(), roots.size())),
47  rq_mat(std::make_pair(qparams.size(), roots.size())), rd_mat(std::make_pair(dparams.size(), roots.size()));
48  for (size_t i = 0; i < roots.size(); ++i) {
49  for (size_t j = 0; j < pparams.size(); ++j) {
50  rp_mat(j, i) = solutions(roots[i], oP + j);
51  }
52  for (size_t j = 0; j < qparams.size(); ++j) {
53  rq_mat(j, i) = solutions(roots[i], oQ + j);
54  }
55  for (size_t j = 0; j < dparams.size(); ++j) {
56  rd_mat(j, i) = solutions(roots[i], oD + j);
57  }
58  }
59  prof->stop();
60  handlers.rp().gemm_outer(rp_mat, cwrap(pparams), params);
61  handlers.rq().gemm_outer(rq_mat, cwrap(qparams), params);
62  handlers.rq().gemm_outer(rd_mat, cwrap(dparams), params);
63 }

77 template <class R>
78 void normalise(const size_t n_roots, const VecRef<R>& params, const VecRef<R>& actions,
79  array::ArrayHandler<R, R>& handler, Logger& logger) {
80  assert(params.size() >= n_roots && actions.size() >= n_roots);
81  for (size_t i = 0; i < n_roots; ++i) {
82  auto dot = handler.dot(params.at(i), params.at(i));
83  dot = std::sqrt(std::abs(dot));
84  if (dot > 1.0e-14) {
85  handler.scal(1. / dot, params.at(i));
86  handler.scal(1. / dot, actions.at(i));
87  } else {
88  logger.msg("solution parameter's length is too small, dot = " + Logger::scientific(dot), Logger::Warn);
89  }
90  }
91 }

102 template <typename T>
103 std::vector<int> select_working_set(const size_t nw, const std::vector<T>& errors, const T threshold,
104  const std::vector<T>& value_errors, const T value_threshold) {
105  auto ordered_errors = std::multimap<T, size_t, std::greater<T>>{};
106  for (size_t i = 0; i < errors.size(); ++i) {
107  if (errors[i] > threshold or (i < value_errors.size() and value_errors[i] > value_threshold))
108  ordered_errors.emplace(errors[i], i);
109  }
110  auto working_set = std::vector<int>{};
111  auto end = (ordered_errors.size() < nw ? ordered_errors.end() : next(begin(ordered_errors), nw));
112  std::transform(begin(ordered_errors), end, std::back_inserter(working_set), [](const auto& el) { return el.second; });
113  std::sort(working_set.begin(), working_set.end());
114  return working_set;
115 }

321  bool solve(const VecRef<R>& parameters, const VecRef<R>& actions, const Problem<R>& problem,
322  bool generate_initial_guess = false) override {
323  if (parameters.empty())
324  throw std::runtime_error("Empty container passed to IterativeSolver::solve()");
325  if (parameters.size() != actions.size())
326  throw std::runtime_error("Inconsistent container sizes in IterativeSolver::solve()");
327  this->m_logger->max_trace_level = Logger::None;
328  if (this->m_verbosity == Verbosity::Detailed) {
329  this->m_logger->max_trace_level = Logger::Info;
330  this->m_logger->data_dump = true;
331  }
332  bool use_diagonals = problem.diagonals(actions.at(0));
333  std::unique_ptr<Q> diagonals;
334  if (use_diagonals)
335  diagonals.reset(new Q{m_handlers->qr().copy(actions.at(0))});
336  if (generate_initial_guess) {
337  if (not use_diagonals)
338  throw std::runtime_error("Default initial guess requested, but diagonal elements are not available");
339  auto guess = m_handlers->qq().select(parameters.size(), *diagonals);
340  size_t root = 0;
341  if (this->m_verbosity >= Verbosity::Summary)
342  molpro::cout << "Initial guess generated from diagonal elements" << std::endl;
343  for (const auto& g : guess) {
344  m_handlers->rp().copy(parameters[root], P{{g.first, 1}});
345  if (this->m_verbosity >= Verbosity::Detailed)
346  molpro::cout << " initial guess " << g.first << std::endl;
347  root++;
348  }
349  }
350  int nwork = parameters.size();
351  std::vector<P> pspace;
352  if (use_diagonals and m_max_p > 0) {
353  auto selectp = m_handlers->qq().select(m_max_p, *diagonals);
354  for (auto s = selectp.begin(); s != selectp.end(); s++)
355  if (s->second > selectp.begin()->second + m_p_threshold) {
356  selectp.erase(s, selectp.end());
357  break;
358  }
359  if (this->m_verbosity >= Verbosity::Summary and not selectp.empty())
360  molpro::cout << selectp.size() << "-dimensional P space selected with threshold "
361  << (m_p_threshold < 1e20 ? std::to_string(m_p_threshold) : "infinity") << " and limit " << m_max_p
362  << std::endl;
363  if (this->m_verbosity >= Verbosity::Detailed)
364  for (const auto& s : selectp)
365  molpro::cout << "P space element " << s.first << " : " << s.second << std::endl;
366  for (const auto& s : selectp)
367  pspace.emplace_back((P){{s.first, 1}});
368  fapply_on_p_type apply_on_p = [&problem](const std::vector<std::vector<value_type>>& pcoeff,
369  const CVecRef<P>& pparams, const VecRef<R>& actions) {
370  problem.p_action(pcoeff, pparams, actions);
371  };
372  auto action_matrix = problem.pp_action_matrix(pspace);
373  nwork = add_p(cwrap(pspace), array::Span<value_type>(action_matrix.data(), action_matrix.size()), parameters,
374  actions, apply_on_p);
375  }
376  for (auto iter = 0; iter < this->m_max_iter && nwork > 0; iter++) {
377  value_type value;
378  if (this->nonlinear()) {
379  value = problem.residual(*parameters.begin(), *actions.begin());
380  nwork = this->add_vector(*parameters.begin(), *actions.begin(), value);
381  } else if (iter > 0 or pspace.empty()) {
382  problem.action(cwrap(parameters.begin(), parameters.begin() + nwork),
383  wrap(actions.begin(), actions.begin() + nwork));
384  nwork = this->add_vector(parameters, actions);
385  }
386  // std::cout << "** nwork="<<nwork<<"use_diagonals="<<use_diagonals<<std::endl;
387  if (nwork > 0) {
388  if (use_diagonals) {
389  m_handlers->rq().copy(parameters.at(0), *diagonals);
390  problem.precondition(wrap(actions.begin(), actions.begin() + nwork), this->working_set_eigenvalues(),
391  parameters.at(0));
392  } else
393  problem.precondition(wrap(actions.begin(), actions.begin() + nwork), this->working_set_eigenvalues());
394  }
395  nwork = this->end_iteration(parameters, actions);
396  if (this->m_verbosity >= Verbosity::Iteration)
397  report();
398  }
399  if (this->m_verbosity == Verbosity::Summary)
400  report();
401  if (this->m_verbosity >= Verbosity::Summary and
402  *std::max_element(m_errors.begin(), m_errors.end()) > m_convergence_threshold)
403  std::cerr << "Solver has not converged to threshold " << m_convergence_threshold << std::endl;
404  return nwork == 0 and *std::max_element(m_errors.begin(), m_errors.end()) <= m_convergence_threshold;
405  }

460  size_t solve_and_generate_working_set(const VecRef<R>& parameters, const VecRef<R>& action) {
461  auto prof = profiler();
462  auto p = prof->push("itsolv::solve_and_generate_working_set");
463  prof->start("itsolv::solve");
464  m_subspace_solver->solve(*m_xspace, n_roots());
465  prof->stop();
466  prof->start("itsolv::temp_solutions");
467  auto nsol = m_subspace_solver->size();
468  std::vector<std::pair<Q, Q>> temp_solutions{};
469  const auto batches = detail::parameter_batches(nsol, parameters.size());
470  for (const auto& batch : batches) {
471  auto [start_sol, end_sol] = batch;
472  auto roots = std::vector<int>(end_sol - start_sol);
473  std::iota(begin(roots), end(roots), start_sol);
474  solution(roots, parameters, action);

481  }

483  }

488  m_convergence_threshold_value);

499  m_handlers->rr().copy(action[i], action[root]);
500  }
501  }
502  }
503  m_logger->msg("add_vector::errors = ", begin(m_errors), end(m_errors), Logger::Trace, 6);
504  return m_working_set.size();
